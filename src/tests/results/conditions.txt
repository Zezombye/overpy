variables {
    global:
        0: A
        1: B
        2: C
        3: D
        4: E
        5: F
        6: G
        7: H
        8: I
        9: J
}
//Optimize for size enabled
rule ("if/else") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Compare(Global.A, ==, Global.B));
            Modify Global Variable(A, Add, True);
        Else;
            Modify Global Variable(B, Add, True);
        End;
        Modify Global Variable(C, Add, True);
    }
}

rule ("if/elif/else") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Compare(Global.A, ==, Global.B));
            Modify Global Variable(A, Add, True);
        Else If(Compare(Global.B, ==, Global.C));
            Modify Global Variable(B, Add, True);
        Else If(Compare(Global.C, ==, Global.D));
            Modify Global Variable(C, Add, True);
        Else;
            Modify Global Variable(D, Add, True);
        End;
        Modify Global Variable(E, Add, True);
    }
}

rule ("nested if/elif/else") {
    event {
        Ongoing - Global;
    }
    actions {
        If(Compare(Global.A, ==, Global.B));
            Modify Global Variable(A, Add, True);
            If(Compare(Global.B, ==, Global.C));
                Modify Global Variable(B, Add, True);
            Else If(Compare(Global.C, ==, Global.D));
                Modify Global Variable(C, Add, True);
            Else;
                Modify Global Variable(D, Add, True);
            End;
        Else If(Compare(Global.B, ==, Global.C));
            Modify Global Variable(B, Add, True);
        Else;
            Modify Global Variable(C, Add, True);
        End;
        Modify Global Variable(D, Add, True);
    }
}

rule ("lone elif/else") {
    event {
        Ongoing - Global;
    }
    actions {
        Else If(Compare(Global.A, ==, Global.B));
            Modify Global Variable(A, Add, True);
        Else If(Compare(Global.B, ==, Global.C));
            Modify Global Variable(B, Add, True);
        Else;
            Modify Global Variable(C, Add, True);
        End;
        Modify Global Variable(D, Add, True);
        Else;
            Modify Global Variable(E, Add, True);
        Else If(Compare(Global.F, ==, Global.G));
            Modify Global Variable(F, Add, True);
        Else;
            Modify Global Variable(G, Add, True);
        End;
        For Global Variable(H, 1, 3, True);
            Modify Global Variable(H, Add, True);
            Else;
                Modify Global Variable(I, Add, True);
            End;
        End;
        Modify Global Variable(J, Add, True);
    }
}

rule ("optimized out empty ifs & if/elif false/true") {
    event {
        Ongoing - Global;
    }
    actions {
        Set Global Variable(A, 2);
        Set Global Variable(A, 2);
        Set Global Variable(B, 3);
        Set Global Variable(C, 6);
        Modify Global Variable(D, Add, True);
        Set Global Variable(A, 2);
        Set Global Variable(B, 4);
        If(Global.A);
            Set Global Variable(B, 6);
        End;
        Modify Global Variable(D, Add, True);
        If(Global.A);
            Modify Global Variable(B, Add, True);
        Else If(Global.C);
            Modify Global Variable(D, Add, True);
        Else;
            Modify Global Variable(E, Add, True);
        End;
        Skip(3);
        If(True);
            Set Global Variable(A, 2);
        Else If(Global.A);
            //lbl_1:
            Set Global Variable(A, 3);
        End;
        Set Global Variable(B, 3);
        Skip(True);
        If(False);
            //lbl_2:
            Set Global Variable(B, 6);
        End;
        Set Global Variable(B, 9);
        Set Global Variable(C, 4);
        Set Global Variable(A, 3);
    }
}
