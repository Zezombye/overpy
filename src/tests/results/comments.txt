variables {
    global:
        0: A
        1: B
        2: C
        3: D
}
rule ("comments") {
    event {
        Ongoing - Global;
    }
    conditions {
        "A"
        Global.A != False;
        "B"
        Global.B != False;
    }
    actions {
        "C++"
        Modify Global Variable(C, Add, True);
        "D++"
        Modify Global Variable(D, Add, True);
        "nothing\r\nsome array"
        Set Global Variable(A, Array(1, 2, 3));
    }
}
