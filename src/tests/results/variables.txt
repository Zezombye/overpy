variables {
    global:
        0: A
        1: foo
        2: bar
        4: E
    player:
        0: A
        1: foo
        2: bar
        4: E
}

//Optimize for size enabled
rule ("variables") {
    event {
        Ongoing - Each Player;
        All;
        All;
    }
    actions {
        Set Global Variable(foo, Add(Add(Global.bar, Global.A), Global.E));
        Set Player Variable(Event Player, foo, Add(Add((Event Player).bar, (Event Player).A), (Event Player).E));
    }
}

rule ("numbers") {
    event {
        Ongoing - Global;
    }
    actions {
        Set Global Variable(A, 74940);
        Set Global Variable(A, -74940);
        Set Global Variable(A, -292);
        Set Global Variable(A, 74940);
        Set Global Variable(A, 292);
        Set Global Variable(A, 1e10);
    }
}
